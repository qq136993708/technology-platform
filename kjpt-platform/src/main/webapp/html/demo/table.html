layui.use(['table'], function(){

  table.render({
    elem: '#tableProject' // 表格元素ID

    ,cols: [[ //表头
      {type: 'checkbox' } // 表格多选
      ,{field: 'id', title: '序号' }
      ,{field: 'username', title: '项目名称', templet: "#Tpl" } // 模版配置列1
      ,{field: 'sex', title: '负责单位', sort: true，, templet: function(d) {

        if (d.id === 1) {
          return '北京';

          return '<span class="text-red"><上海/span>';
        }
        }
      }
      ,{field: 'city', title: '专业类型'} 
      ,{field: 'sign', title: '立项年度'}
    ]],
    method: 'GET', // 数据请求接口类型 默认 get, 如果是get 可以去掉此项
    where: {
      useID: 'admin' // 表格请求时传递的参数
      //  默认请求参数 页码 => page, 每页条数 => limit
    },
    ,request: {
      page: 'pageIndex', // 重置默认分页请求请求参数 page => pageIndex
      limit: 'pageSize' // 重置默认分页请求请求参数 limit => pageSize

    ,page: true //开启分页
